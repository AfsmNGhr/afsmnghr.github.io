Rails Project with Multiple Databases

May 16, 2015 • rails • nix • ruby • db •

Потребовалось для проекта Free Freight очищать не актуальные позиции.
Решено было сделать базу archive, доступную из rails enviroment …

В интернете много способов таких соединений,
но попался весьма интересный способ …
Connecting Rails Project With Multiple Databases

  • Archive base for establishing connection

class Archive::ArchiveBase

  def archive_connection(model_name)
    connection(model_name)
  end

  private

  def connection(model_name)
    # self.abstract_class = true
    databases = YAML::load(IO.read('config/archive.yml'))
    model_name.establish_connection(databases[Rails.env])
  end
end

  • Archive rake

# coding: utf-8
namespace :archive do
  desc 'Configure the variables that rails need in order to look up for the db
configuration in a different folder'
  task :set_custom_db_config_paths do
    ENV['SCHEMA'] = 'archive/schema.rb'
    Rails.application.config.paths['db'] = ['archive']
    Rails.application.config.paths['db/migrate'] = ['archive/migrate']
    Rails.application.config.paths['db/seeds.rb'] = ['archive/seeds.rb']
    Rails.application.config.paths['config/database'] = ['config/archive.yml']
  end

  namespace :db do
    task drop: :set_custom_db_config_paths do
      Rake::Task['db:drop'].invoke
    end

     task drop: :set_custom_db_config_paths do
      Rake::Task['db:setup'].invoke
    end

    task create: :set_custom_db_config_paths do
      Rake::Task['db:create'].invoke
    end

    task migrate: :set_custom_db_config_paths do
      Rake::Task['db:migrate'].invoke
    end

    task rollback: :set_custom_db_config_paths do
      Rake::Task['db:rollback'].invoke
    end

    task seed: :set_custom_db_config_paths do
      Rake::Task['db:seed'].invoke
    end

    namespace :test do
      task prepare: :set_custom_db_config_paths do
        Rake::Task['db:test:prepare'].invoke
      end
    end

    task version: :set_custom_db_config_paths do
      Rake::Task['db:version'].invoke
    end
  end
end

  • Archive folder

Создаём archive/migrate и archive/seed.rb

  • Archive yml

В config/archive.yml настраиваем параметры окружений для соединения

  • Use rake archive task

Используем новый namespace

rake archive:db:create
rake archive:db:migrate
rake archive:db:seed

  • Model position

Для полноценной работы нужны таблицы в двух namespaces

class Archive::Position < Position
  Archive::ArchiveBase.new.archive_connection(self)
end

  • Backup

namespace :backup do
  desc 'move to archive old positions'
  task positions: :environment do
    ActiveRecord::Base.connection.execute(<<-EOQ)
INSERT IGNORE INTO `free_freight_#{Rails.env}_archive`.`positions`
SELECT *
FROM `free_freight_#{Rails.env}`.`positions`
WHERE `laycan_last_day` < '#{Date.today}';
EOQ

    Position.where('laycan_last_day < ?', Date.today).delete_all
  end
end

добавляем в whenever

every 1.day, at: '22:45 PM' do
  rake 'backup:positions'
end

  • Результат

Плюсы:

  • Актуальные позиции
  • Скорость возросла за счёт уменьшения позиций
  • Весь archive доступен для различных манипуляций

Минусы:

  • Нет миграций для archive
    (можно генерить в db namespace и перебрасывать в archive)

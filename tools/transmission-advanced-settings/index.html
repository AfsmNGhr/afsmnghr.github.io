Transmission daemon (transmission)

  • Добавляем пользователя в группу transmission
sudo usermod -aG $USER transmission
  • Меняем группу у всех папок, c которыми будет взаимодействовать transmission
sudo chown -R $USER:transmission $PATH_TORRENTS
  • Включаем bash для transmission
transmission:x:169:169:...:/bin/bash
  • Разрешаем выключение (/etc/sudoers)
transmission ALL=(ALL) NOPASSWD: /usr/bin/systemctl poweroff
  • Передаём переменные для запуска script`a . Подробнее
sudo mkdir /etc/systemd/system/transmission.service.d
# cat /etc/systemd/system/transmission.service.d/custom.conf

[Service]
Environment=DISPLAY=:0
EnvironmentFile=/etc/systemd/system/transmission.service.d/variables.conf

# cat /etc/systemd/system/transmission.service.d/variables.conf
PATH=$(echo $PATH)
SHELL=/bin/bash
GEM_HOME=$(gem environment gemdir)
GEM_PATH=$(gem environment gemdir)

Transmission daemon (user)

  • Меняем пользователя и передаём переменные
# cat /etc/systemd/system/transmission.service.d/custom.conf

[Service]
User=user
Environment=DISPLAY=:0
# см. выше
EnvironmentFile=/etc/systemd/system/transmission.service.d/variables.conf

Torrent

#!/usr/bin/env ruby
# coding: utf-8
require 'fest'

def delete_torrents_done
  @ids = `transmission-remote -l | awk '{ print $1 }`.split("\n")
  @ids -= [@ids[0], @ids[-1]]

  @ids.each do |id|
    system("transmission-remote -t '#{id}' -r")
  end
end

def check_statuses
  @statuses = `transmission-remote -l | awk '{ print $5 }'`.split("\n")
  @statuses -= [@statuses[0], @statuses[-1]]
end

if `cat /sys/class/backlight/intel_backlight/brightness`.to_i == 0
  check_statuses
  counts = 0

  @statuses.each do |status|
    counts += 1 if status == 'Done'
  end

  sleep 600
  if counts == @statuses.size
    delete_torrents_done
    system('poweroff')
  end
else
  @fest = Fest.new
  @fest.say('Загр+узка завершена')
end

else
sudo systemctl start transmission
fi }

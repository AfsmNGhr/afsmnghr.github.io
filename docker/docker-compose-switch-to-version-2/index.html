Docker-compose switch to version 2

Feb 27, 2016 • docker-compose • docker • nix •

Совсем недавно, после выхода Compose 1.6.0+ и Docker 1.10.0+, стала доступна 2 версия.
Добавлены ключи: services, volumes, networks. Будут удалены ключи: links.

Version 1

app:
  build: .
  ports:
    - '3000:3000'
  volumes:
    - .:/home/web
  links:
    - db
db:
  image: sameersbn/postgresql
  ports:
    - '5432'
  volumes_from:
    - db_data
  env_file:
    - $HOME/.docker/$APP/db.env
db_data:
  image: tianon/true
  volumes:
    - /var/lib/postgresql

Version 2 (fallback mode)

Пока links работают по-старому только network_mode: bridge.

version: '2'
services:
  app:
    build: .
    ports:
      - '3000:3000'
    volumes:
      - .:/home/web
    depends_on:
      - db
    links:
      - db
    network_mode: bridge
  db:
    image: sameersbn/postgresql
    ports:
      - '5432'
    volumes:
      - db_data:/var/lib/postgresql
    env_file:
      - $HOME/.docker/$APP/db.env
    network_mode: bridge
volumes:
  db_data:

Version 2 (without links, but use env file)

version: '2'
services:
  app:
    build: .
    ports:
      - '3000:3000'
    volumes:
      - .:/home/web
